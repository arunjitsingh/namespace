  6 /**
  7  * Creates a namespace from a '.'-separated string and an optional parent
  8  * object.
  9  * @param {String} ns The namespace to create.
 10  * @param {Object=} parent The parent object. Defaults to {this}, whatever it
 11  *     may be.
 12  */
 13 function namespace(ns, parent) {
 14   parent = parent || this;
 15   var parts = ns.split('.'),
 16     ref = parent;
 17 
 18   parts.forEach(function(part) {
 19     ref[part] = ref[part] || {};
 20     ref = ref[part];
 21   });
 22 }
 23 
 24 /**
 25  * @this {window|module.exports}
 26  * @suppress {globalThis}
 27  */
 28 this['namespace'] = namespace;
